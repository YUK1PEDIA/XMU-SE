package com.xmu.aftersale.dao;

import com.xmu.aftersale.mapper.jpa.AfterSalePackagePoMapper;
import com.xmu.aftersale.mapper.po.AfterSalePackagePo;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.context.config.annotation.RefreshScope;
import org.springframework.stereotype.Repository;

import java.time.LocalDateTime;

@RefreshScope
@Slf4j
@Repository
public class AfterSalePackageDao {

    @Autowired
    private AfterSalePackagePoMapper afterSalePackagePoMapper;

    // 创建新运单
    public void createAfterSalePackage(Long orderId, Byte type) {
        Long packageId = System.currentTimeMillis() % 1000L; // 随机生成运单ID
        AfterSalePackagePo afterSalePackagePo = new AfterSalePackagePo();
        afterSalePackagePo.setOrderId(orderId);
        afterSalePackagePo.setType(type);
        afterSalePackagePo.setPackageId(packageId); // 确保 packageId 不为 null
        afterSalePackagePoMapper.save(afterSalePackagePo);
    }

}
