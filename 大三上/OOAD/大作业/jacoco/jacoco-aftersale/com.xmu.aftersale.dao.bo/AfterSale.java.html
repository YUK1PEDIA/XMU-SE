package com.xmu.aftersale.dao.bo;

import com.xmu.aftersale.controller.dto.EditAfterSaleDto;
import com.xmu.aftersale.dao.AfterSaleDao;
import lombok.Data;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

@Data
@Component
public class AfterSale {

    private Byte type; // 售后类型

    private Long orderId; // 订单ID

    private Byte status; // 售后单状态：0 待审核；1 已审核

    @Autowired
    private AfterSaleDao afterSaleDao;

    public void update(EditAfterSaleDto editAfterSaleDto, Long afterSaleId) {
        afterSaleDao.updateAfterSaleInfo(editAfterSaleDto, afterSaleId);
    }
}
