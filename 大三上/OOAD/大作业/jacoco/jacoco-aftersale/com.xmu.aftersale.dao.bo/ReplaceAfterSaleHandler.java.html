package com.xmu.aftersale.dao.bo;

import com.xmu.aftersale.dao.AfterSalePackageDao;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

@Component("replaceAfterSale")
public class ReplaceAfterSaleHandler extends AfterSaleHandler {

    @Autowired
    private AfterSalePackageDao afterSalePackageDao;

    @Override
    public void reviewAfterSale(Long orderId) {
        afterSalePackageDao.createAfterSalePackage(orderId, (byte) 1);
    }
}
