package com.xmu.aftersale.util;

import com.xmu.aftersale.dao.bo.AfterSale;
import com.xmu.aftersale.mapper.po.AfterSalePo;
import com.xmu.aftersale.mapper.po.AfterSaleOrderPo;

public class CloneFactory {

    /**
     * 从 AfterSalePo 转换为 AfterSale
     */
    public static void copy(AfterSalePo source, AfterSale target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }

        target.setType(source.getType());
        target.setOrderId(source.getOrderId());
        target.setStatus(source.getStatus());
    }

    /**
     * 从 AfterSale 转换为 AfterSalePo
     */
    public static void copy(AfterSale source, AfterSalePo target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }

        target.setType(source.getType());
        target.setOrderId(source.getOrderId());
        target.setStatus(source.getStatus());
    }


}
