package com.xmu.aftersale.util;

import com.xmu.aftersale.dao.bo.AfterSaleHandler;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.Map;

@Component
public class AfterSaleFactory {
    private final Map<Byte, AfterSaleHandler> afterSaleHandlerMap;

    /**
     * 构造函数注入，初始化处理器映射表
     * @param afterSaleHandlerMap
     */
    @Autowired
    public AfterSaleFactory(Map<String, AfterSaleHandler> afterSaleHandlerMap) {
        this.afterSaleHandlerMap = Map.of(
                (byte) 1, afterSaleHandlerMap.get("refundAfterSale"),
                (byte) 2, afterSaleHandlerMap.get("replaceAfterSale")
        );
    }

    /**
     * 根据售后类型获取具体的售后实例
     *
     * @param type 售后类型
     * @return 售后实例
     */
    public AfterSaleHandler getAfterSaleHandler(Byte type) {
        AfterSaleHandler afterSaleHandler = afterSaleHandlerMap.get(type);
        if (afterSaleHandler == null) {
            throw new IllegalArgumentException("未知的售后类型：" + type);
        }
        return afterSaleHandler;
    }
}
