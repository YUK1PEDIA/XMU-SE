package com.xmu.service.util;

import com.xmu.service.dao.bo.*;
import com.xmu.service.mapper.po.*;

public class CloneFactory {

    public static void copy(ServiceContractPo source, ServiceContract target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }
        target.setDetails(source.getDetails());
        target.setBeginTime(source.getBeginTime());
        target.setEndTime(source.getEndTime());
    }

    public static void copy(ServicePo source, ServiceBo target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }
        target.setName(source.getName());
        target.setDescription(source.getDescription());
        target.setService_type(source.getServiceType());
    }

    public static void copy(ServiceContract source, ServiceContractPo target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }

        target.setServiceId(source.getServiceId());
        target.setContractId(source.getContractId());
        target.setShopId(source.getShopId());
        target.setServiceOrderId(source.getServiceOrderId());
        target.setDetails(source.getDetails());
        target.setBeginTime(source.getBeginTime());
        target.setEndTime(source.getEndTime());
    }

    public static void copy(ServiceOrderPo source, ServiceOrder target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }

        target.setId(source.getId());
        target.setType(source.getType());
        target.setStatus(source.getStatus());
        target.setAddress(source.getAddress());
        target.setDescription(source.getDescription());
        target.setResult(source.getResult());
    }

    public static void copy(ServiceOrder source, ServiceOrderPo target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or Target cannot be null");
        }

        // 根据字段名进行赋值
        target.setId(source.getId());
        target.setType(source.getType());
        target.setStatus(source.getStatus());
        target.setAddress(source.getAddress());
        target.setDescription(source.getDescription());
        target.setResult(source.getResult());
    }

    public static void copy(ServicePackagePo source, ServicePackage target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source and target must not be null");
        }

        target.setId(source.getId());
        target.setType(source.getType());
        target.setServiceOrderId(source.getServiceOrderId());
    }

    public static void copy(ServicePackage source, ServicePackagePo target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source and target must not be null");
        }

        target.setId(source.getId());
        target.setType(source.getType());
        target.setServiceOrderId(source.getServiceOrderId());
    }

    public static void copy(ServiceProdOrder source, ServiceProdOrderPo target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or target cannot be null.");
        }

        // 复制字段
        target.setServiceOrderId(source.getServiceOrderId());
        target.setAddress(source.getAddress());
        target.setDescription(source.getDescription());
        target.setCreateTime(source.getCreateTime());
    }

    public static void copy(ServiceProdOrderPo source, ServiceProdOrder target) {
        if (source == null || target == null) {
            throw new IllegalArgumentException("Source or target cannot be null.");
        }

        // 复制字段
        target.setServiceOrderId(source.getServiceOrderId());
        target.setAddress(source.getAddress());
        target.setDescription(source.getDescription());
        target.setCreateTime(source.getCreateTime());
    }
}
