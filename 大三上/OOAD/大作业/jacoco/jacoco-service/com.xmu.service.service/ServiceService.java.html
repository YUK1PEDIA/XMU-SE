package com.xmu.service.service;

import com.xmu.service.controller.dto.ServiceContractDto;
import com.xmu.service.dao.ServiceContractDao;
import com.xmu.service.dao.ServiceDao;
import com.xmu.service.dao.bo.ServiceBo;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Propagation;
import org.springframework.transaction.annotation.Transactional;

@Service
@Transactional(propagation = Propagation.REQUIRED)
@RequiredArgsConstructor // 自动生成构造函数
@Slf4j
public class ServiceService {

    private final ServiceDao serviceDao; // final 字段
    private final ServiceContractDao serviceContractDao; // final 字段

    /**
     * 定义服务合同
     * @param serviceId
     * @param serviceContractDto
     * @param shopId
     */
    public void createContract(Long serviceId, ServiceContractDto serviceContractDto, Long shopId) {
        // 先通过 serviceId 查找出 service 对象
        ServiceBo serviceBo = serviceDao.findServiceId(serviceId);

        // 将创建服务合同的职责分配给 service 对象
        serviceBo.createServiceContract(serviceContractDto, serviceId, shopId, serviceContractDao);
    }
}
