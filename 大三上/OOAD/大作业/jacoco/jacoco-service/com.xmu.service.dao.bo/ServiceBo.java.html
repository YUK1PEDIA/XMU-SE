package com.xmu.service.dao.bo;

import com.xmu.service.controller.dto.ServiceContractDto;
import com.xmu.service.dao.ServiceContractDao;
import lombok.Data;
import lombok.NoArgsConstructor;
import lombok.extern.slf4j.Slf4j;

import java.time.LocalDateTime;

@Data
@NoArgsConstructor
@Slf4j
public class ServiceBo {

    private String name;

    private String description;

    private String service_type;

    public void createServiceContract(ServiceContractDto serviceContractDto, Long serviceId, Long shopId, ServiceContractDao serviceContractDao) {
        String details = serviceContractDto.getDetails();
        Long serviceProviderId = Long.parseLong(serviceContractDto.getServiceProviderId());

        ServiceContract serviceContract = new ServiceContract(); // 创建新的服务合同对象
        serviceContract.setDetails(details);
        serviceContract.setServiceOrderId(serviceProviderId);
        serviceContract.setBeginTime(LocalDateTime.now());
        serviceContract.setEndTime(LocalDateTime.now().plusDays(7)); // 设置合同有效期为 7 天
        serviceContract.setServiceId(serviceId);
        serviceContract.setShopId(shopId);

        Long contractId = System.currentTimeMillis() % 1000L; // 根据时间戳随机生成合同ID
        serviceContract.setContractId(contractId);

        serviceContractDao.insertServiceContract(serviceContract);
    }
}
