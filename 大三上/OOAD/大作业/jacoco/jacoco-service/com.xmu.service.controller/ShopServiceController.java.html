package com.xmu.service.controller;

import cn.edu.xmu.javaee.core.aop.LoginUser;
import cn.edu.xmu.javaee.core.model.ReturnNo;
import cn.edu.xmu.javaee.core.model.ReturnObject;
import cn.edu.xmu.javaee.core.model.dto.UserDto;
import com.xmu.service.controller.dto.ServiceContractDto;
import com.xmu.service.service.ServiceOrderService;
import com.xmu.service.service.ServiceService;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequiredArgsConstructor
@RequestMapping(value = "/shops/{shopId}", produces = "application/json;charset=UTF-8")
@Slf4j
public class ShopServiceController {

    @Autowired
    private ServiceService serviceService;
    @Autowired
    private ServiceOrderService serviceOrderService;

    /**
     * 定义服务合同
     * @param shopId
     * @param id
     * @param dto
     * @return
     */
    @PostMapping("/service/{id}/createContract")
    public ReturnObject createContract(@PathVariable("shopId") Long shopId,
                                       @PathVariable("id") Long id, // 服务 id
                                       @RequestBody ServiceContractDto dto
                                       /*@LoginUser UserDto user*/) {
        serviceService.createContract(id, dto, shopId);
        return new ReturnObject(ReturnNo.CREATED);
    }

    /**
     * 取消服务单
     * @param shopId
     * @param id
     * @return
     */
    @DeleteMapping("/service/{id}")
    public ReturnObject cancelServiceOrder(@PathVariable("shopId") Long shopId,
                                           @PathVariable("id") Long id
                                           /*@LoginUser UserDto user*/) { // 服务 id
        serviceOrderService.cancelService(id);
        return new ReturnObject(ReturnNo.OK);
    }
}
