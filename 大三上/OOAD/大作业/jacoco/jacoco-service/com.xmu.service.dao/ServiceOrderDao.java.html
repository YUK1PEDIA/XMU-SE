package com.xmu.service.dao;

import com.xmu.service.dao.bo.ServiceOrder;
import com.xmu.service.mapper.jpa.ServiceOrderPoMapper;
import com.xmu.service.mapper.po.ServiceOrderPo;
import com.xmu.service.mapper.po.ServicePackagePo;
import com.xmu.service.util.CloneFactory;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.context.config.annotation.RefreshScope;
import org.springframework.stereotype.Repository;

import java.util.Optional;

@RefreshScope
@Slf4j
@Repository
public class ServiceOrderDao {

    @Autowired
    private ServiceOrderPoMapper serviceOrderPoMapper;

    public ServiceOrder findServiceOrderById(Long serviceOrderId) {
        Optional<ServiceOrderPo> serviceOrderPoOptional = serviceOrderPoMapper.findById(serviceOrderId);

        if (serviceOrderPoOptional.isPresent()) {
            ServiceOrderPo serviceOrderPo = serviceOrderPoOptional.get(); // 从 Optional 中获取实际对象
            ServiceOrder serviceOrder = new ServiceOrder();
            CloneFactory.copy(serviceOrderPo, serviceOrder);
            return serviceOrder;
        } else {
            throw new RuntimeException("ServiceOrder with ID " + serviceOrderId + " not found.");
        }
    }


}
