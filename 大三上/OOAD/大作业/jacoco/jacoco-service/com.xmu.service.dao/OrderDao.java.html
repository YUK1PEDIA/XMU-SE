package com.xmu.service.dao;

import com.xmu.service.dao.bo.ServiceProdOrder;
import com.xmu.service.mapper.jpa.ServiceProdOrderMapper;
import com.xmu.service.mapper.po.ServiceProdOrderPo;
import com.xmu.service.util.CloneFactory;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.context.config.annotation.RefreshScope;
import org.springframework.stereotype.Repository;

@RefreshScope
@Slf4j
@Repository
public class OrderDao {

    @Autowired
    private ServiceProdOrderMapper serviceProdOrderMapper;

    /**
     * 创建订单
     * @param serviceProdOrder
     */
    public void createOrder(ServiceProdOrder serviceProdOrder) {
        ServiceProdOrderPo serviceProdOrderPo = new ServiceProdOrderPo();
        CloneFactory.copy(serviceProdOrder, serviceProdOrderPo);
        serviceProdOrderMapper.save(serviceProdOrderPo);
    }
}
