package com.xmu.service.dao;

import com.xmu.service.dao.bo.ServiceContract;
import com.xmu.service.mapper.jpa.ServiceContractPoMapper;
import com.xmu.service.mapper.po.ServiceContractPo;
import com.xmu.service.util.CloneFactory;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cloud.context.config.annotation.RefreshScope;
import org.springframework.stereotype.Repository;


@RefreshScope
@Slf4j
@Repository
public class ServiceContractDao {

    @Autowired
    private ServiceContractPoMapper serviceContractPoMapper;

    /**
     * 向数据库插入新增的服务合同
     * @param serviceContract
     */
    public void insertServiceContract(ServiceContract serviceContract) {
        // 将服务合同对象转化成 po 对象，便于存进数据库
        ServiceContractPo serviceContractPo = new ServiceContractPo();
        CloneFactory.copy(serviceContract, serviceContractPo);
        serviceContractPoMapper.save(serviceContractPo);
    }
}
